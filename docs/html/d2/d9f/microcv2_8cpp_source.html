144 if (contours.size() == 0) {
145 return false;
146 }
147
148 // Sort the contours to put the largest at index 0
149 std::sort(contours.begin(), contours.end(), [](const contour_t& a, const contour_t& b) {

151 });

157 cv::Point leftTop = contours[0][0];

159 for (const auto& pt : contours[0]) {
160 // bottom left point
161 if (pt.y > bottomLeft.y || (bottomLeft.y == pt.y && pt.x < bottomLeft.x)) {
162 bottomLeft = pt;
163 }
164 // left top point
165 if (pt.x < leftTop.x || (pt.x == leftTop.x && pt.y < leftTop.y)) {

167 }

169

172 float y_intercept = leftTop.y - slope * leftTop.x;
173
174 // Draw the line
175 int16_t p1_x, p1_y, p2_x, p2_y;
176 p1_y = 0;
177 p2_y = mask.rows - 1;
178 p1_x = (p1_y - y_intercept) / slope;
179 p2_x = (p2_y - y_intercept) / slope;
180 cv::line(centerLine, cv::Point(p1_x, p1_y), cv::Point(p2_x, p2_y), cv::Scalar(255), 1);
181
182 // Point where the slope line intersects the WHITE_VERTICAL_CROP line
183 cv::Point intersectionPoint;
184 intersectionPoint.y = Params::WHITE_VERTICAL_CROP;
185 intersectionPoint.x = (intersectionPoint.y - y_intercept) / slope;
186 cv::line(centerLine, cv::Point(intersectionPoint.x, 0), cv::Point(intersectionPoint.x, mask.rows - 1), cv::Scalar(255), 1);
187
188 dist = intersectionPoint.x - Params::WHITE_CENTER_POS;

190 uint8_t MAX_WHITE_DIST = CLAMP_CENTER_POS(IMG_COLS, Params::WHITE_CENTER_POS);
191
192 if (dist > MAX_WHITE_DIST) dist = MAX_WHITE_DIST;

200{
