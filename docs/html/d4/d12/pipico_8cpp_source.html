58esp_err_t PiPico::receivePacket()
59{
60 // check if there are enough buffered bytes for the sync bytes + packet
61 constexpr uint8_t expectedBytes = sizeof(SYNC_BYTES) + sizeof(PicoToEspPacket);
62 size_t buffSize;
63
64 uart_get_buffered_data_len(UART_NUM, &buffSize);
65 if (buffSize < expectedBytes) {
66 return ESP_FAIL;
67 }
68
69 // Check for the sync bytes
70 bool syncFound = false;
71 uint32_t syncBuffer = 0;
72 while (!syncFound) {
73 uint8_t byte;
74 uart_read_bytes(UART_NUM, &byte, 1, 20 / portTICK_PERIOD_MS);
75 syncBuffer = (syncBuffer >> 8) | (byte << 24);
76 if (syncBuffer == SYNC_BYTES) {
77 syncFound = true;
78 }
79 vTaskDelay(1);
80 }
81

85 if (length == sizeof(recvPacket)) {
